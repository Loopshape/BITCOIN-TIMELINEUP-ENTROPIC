// Bitcoin timestamp entropy generation
async function generateEntropy(timestamp) {
  const hash = await sha256(timestamp);
  const entropy = hash.slice(0, 16); // First 64 bits of the hash
  return "0x" + entropy.toUpperCase();
}

async function sha256(data) {
  // Real cryptographic hash using Web Crypto API
  const msgBuffer = new TextEncoder().encode(data);
  const hashBuffer = await crypto.subtle.digest('SHA-256', msgBuffer);
  const hashArray = Array.from(new Uint8Array(hashBuffer));
  const hashHex = hashArray.map(b => b.toString(16).padStart(2, '0')).join('');
  return hashHex;
}
